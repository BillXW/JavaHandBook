#include<iostream>
#include<vector>
using namespace std;
//冒泡 
void bubble_sort(vector<int> &nums)
{
	cout << nums.size() << endl;
    for (int i = 0; i < nums.size() - 1; i++) { // times
        for (int j = 0; j < nums.size() - i - 1; j++) { // position
            if (nums[j] > nums[j + 1]) {
                int temp = nums[j];
                nums[j] = nums[j + 1];
                nums[j + 1] = temp;
            }
        }
    }
}
//插入 
void insert_sort(vector<int> &nums)
{
    for (int i = 1; i < nums.size(); i++) { // position
        for (int j = i; j > 0; j--) {
            if (nums[j] < nums[j - 1]) {
                int temp = nums[j];
                nums[j] = nums[j - 1];
                nums[j - 1] = temp;
            }
        }
    }
}

//选择 
void selection_sort(vector<int> &nums)
{
    for (int i = 0; i < nums.size(); i++) { // position
        int min = i;
        for (int j = i + 1; j < nums.size(); j++) {
            if (nums[j] < nums[min]) {
                min = j;
            }
        }

        int temp = nums[i];
        nums[i] = nums[min];
        nums[min] = temp;
    }
}
int main(){
	vector<int> num;
	num.push_back(5);
	num.push_back(1);
	num.push_back(4);
	num.push_back(9);
	insert_sort(num);
		cout << num.size() << endl;
	for(int i=0;i<num.size();i++){
		cout << num[i] << endl;
	}
	return 0;
}
